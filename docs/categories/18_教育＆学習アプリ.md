```swift
// 学習進捗モデル
struct LearningProgress: Codable {
    var lessonsCompleted: Set<String>
    var quizScores: [String: Int]
    var streakDays: Int
    var lastStudyDate: Date

    var completionPercentage: Double {
        Double(lessonsCompleted.count) / Double(totalLessons)
    }
}

// 進捗管理
class ProgressManager: ObservableObject {
    @Published var progress: LearningProgress

    func completeLesson(_ id: String) {
        progress.lessonsCompleted.insert(id)
        updateStreak()
        save()
    }

    func updateStreak() {
        let calendar = Calendar.current
        if calendar.isDateInToday(progress.lastStudyDate) {
            return
        } else if calendar.isDateInYesterday(progress.lastStudyDate) {
            progress.streakDays += 1
        } else {
            progress.streakDays = 1
        }
        progress.lastStudyDate = Date()
    }
}
```

### スペーシング復習（SRS）

```swift
// Spaced Repetition System
struct FlashCard: Identifiable {
    let id: UUID
    var front: String
    var back: String
    var easeFactor: Double = 2.5
    var interval: Int = 1
    var repetitions: Int = 0
    var nextReviewDate: Date

    mutating func review(quality: Int) {
        // SM-2 Algorithm
        if quality >= 3 {
            switch repetitions {
            case 0:
                interval = 1
            case 1:
                interval = 6
            default:
                interval = Int(Double(interval) * easeFactor)
            }
            repetitions += 1
            easeFactor = max(1.3, easeFactor + 0.1 - Double(5 - quality) * (0.08 + Double(5 - quality) * 0.02))
        } else {
            repetitions = 0
            interval = 1
        }
        nextReviewDate = Calendar.current.date(byAdding: .day, value: interval, to: Date())!
    }
}
```

### クイズシステム

```swift
// クイズ管理
struct Quiz {
    let questions: [Question]
    var currentIndex = 0
    var score = 0

    struct Question {
        let text: String
        let options: [String]
        let correctIndex: Int
    }

    mutating func answer(_ selectedIndex: Int) -> Bool {
        let isCorrect = selectedIndex == questions[currentIndex].correctIndex
        if isCorrect { score += 1 }
        currentIndex += 1
        return isCorrect
    }
}

struct QuizView: View {
    @State var quiz: Quiz
    @State var selectedAnswer: Int?
    @State var showResult = false

    var body: some View {
        VStack {
            // 進捗
            ProgressView(value: Double(quiz.currentIndex), total: Double(quiz.questions.count))

            // 質問
            Text(quiz.questions[quiz.currentIndex].text)
                .font(.title2)
                .padding()

            // 選択肢
            ForEach(quiz.questions[quiz.currentIndex].options.indices, id: \.self) { index in
                Button(action: { selectedAnswer = index }) {
                    Text(quiz.questions[quiz.currentIndex].options[index])
                        .frame(maxWidth: .infinity)
                        .padding()
                        .background(selectedAnswer == index ? Color.blue : Color.gray.opacity(0.2))
                        .cornerRadius(10)
                }
            }

            // 回答ボタン
            Button("回答") {
                if let answer = selectedAnswer {
                    _ = quiz.answer(answer)
                    selectedAnswer = nil
                }
            }
            .disabled(selectedAnswer == nil)
        }
    }
}
```

## ResearchKit統合

### 研究同意取得

```swift
import ResearchKit

class ConsentTask {
    static func create() -> ORKOrderedTask {
        let document = ORKConsentDocument()
        document.title = "研究への同意"

        // セクション追加
        let sections = [
            ORKConsentSection(type: .overview),
            ORKConsentSection(type: .dataGathering),
            ORKConsentSection(type: .privacy),
            ORKConsentSection(type: .studyTasks)
        ]
        document.sections = sections

        // 署名
        let signature = ORKConsentSignature(
            forPersonWithTitle: "参加者",
            dateFormatString: nil,
            identifier: "participant"
        )
        document.addSignature(signature)

        // ステップ
        let visualStep = ORKVisualConsentStep(identifier: "visual", document: document)
        let reviewStep = ORKConsentReviewStep(
            identifier: "review",
            signature: signature,
            in: document
        )

        return ORKOrderedTask(identifier: "consent", steps: [visualStep, reviewStep])
    }
}
```
